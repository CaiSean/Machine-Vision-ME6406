Stage 1: Radial alignment constraint

for i = 1:length(X_W)
    A(i, :) = [X_W(i)*Y_u(i), Y_W(i)*Y_u(i),...
               -X_W(i)*X_u(i), -Y_W(i)*X_u(i), Y_u(i)];

    b(i, :) = X_u(i);
end

x = A\b;

r11_ = x(1); r12_ = x(2);
r21_ = x(3); r22_ = x(4);
tx_ = x(5);

clear A b x

Sr = r11_^2 + r12_^2 + r21_^2 + r22_^2;

diff = r11_*r22_ - r12_*r21_;

if diff == 0
    ty = sqrt(1/Sr);
else
    ty = sqrt((Sr-sqrt(Sr^2-4*diff^2))/(2*diff^2));
end

clear diff

Check ty sign

Check S1, S2 signs
